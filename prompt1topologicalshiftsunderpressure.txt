### Embedding Extraction (Qwen2.5-specific):
```python
def extract_embeddings(model, tokenizer, text, layer_idx=14):
    """
    Extract embeddings from Qwen2.5 at specified layer.
    
    Args:
        model: Qwen2.5 model instance
        tokenizer: Qwen2.5 tokenizer
        text: Input text (already formatted with chat template)
        layer_idx: Which layer to extract (0-23 for 0.5B model)
    
    Returns:
        numpy array of embeddings (num_tokens, hidden_dim)
    """
    inputs = tokenizer(text, return_tensors="pt")
    
    with torch.no_grad():
        outputs = model(**inputs, output_hidden_states=True)
    
    # outputs.hidden_states is a tuple of (num_layers + 1) tensors
    # Index 0 is embeddings, indices 1-24 are layer outputs
    # We want layer_idx (1-indexed in the tuple)
    hidden_state = outputs.hidden_states[layer_idx + 1]  # +1 because index 0 is embeddings
    
    # Shape: (batch_size, seq_len, hidden_dim)
    # Remove batch dimension and convert to numpy
    embeddings = hidden_state.squeeze(0).cpu().numpy()
    
    return embeddings
```

### For generation with hidden states:
```python
def generate_with_embeddings(model, tokenizer, prompt, layer_idx=14, max_new_tokens=100):
    """
    Generate response and extract embeddings from the generation process.
    """
    messages = [{"role": "user", "content": prompt}]
    text = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)
    inputs = tokenizer(text, return_tensors="pt")
    
    # Generate
    with torch.no_grad():
        outputs = model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            output_hidden_states=True,
            return_dict_in_generate=True,
            do_sample=True,
            temperature=0.7
        )
    
    # Extract response text
    response = tokenizer.decode(outputs.sequences[0][inputs.input_ids.shape[1]:], skip_special_tokens=True)
    
    # For embeddings, run a forward pass on the full generated sequence
    with torch.no_grad():
        full_outputs = model(outputs.sequences, output_hidden_states=True)
    
    embeddings = full_outputs.hidden_states[layer_idx + 1].squeeze(0).cpu().numpy()
    
    return response, embeddings
```
